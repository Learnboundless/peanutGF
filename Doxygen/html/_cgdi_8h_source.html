263  void LineWithArrow(Vector2D from, Vector2D to, double size) {
264  Vector2D norm = Vec2DNormalize(to - from);
265 
266  //计算我们的箭头朝向在哪里
267  Vector2D CrossingPoint = to - (norm * size);
268 
269  //计算箭头的两个点
270  Vector2D ArrowPoint1 = CrossingPoint + (norm.Perp() * 0.4f * size);
271  Vector2D ArrowPoint2 = CrossingPoint - (norm.Perp() * 0.4f * size);
272 
273  //绘制封闭线
274  MoveToEx(m_hdc, (int)from.x, (int)from.y, NULL);
275  LineTo(m_hdc, (int)CrossingPoint.x, (int)CrossingPoint.y);
276 
277  //绘制箭头
278  POINT p[3];
279 
280  p[0] = VectorToPOINT(ArrowPoint1);
281  p[1] = VectorToPOINT(ArrowPoint2);
282  p[2] = VectorToPOINT(to);
283 
284  SetPolyFillMode(m_hdc, WINDING);
285  Polygon(m_hdc, p, 3);
286  }
